        \begin{itemize}
          \item \textbf{Reinvent} − SMILES-level RL
          \item \textbf{LibInvent} − R-group decoration

      \end{block}

        \begin{itemize}
          \item Custom QSAR scorer (\texttt{reinvent\_qsar\_pharm\_score.py})
          \item Tanimoto similarity to Series C/E seeds
          \item QSAR predicted \pIC{} threshold filter
          \item Pharmacophore screen (Phase)
        \end{itemize}
      \end{block}

\section{Challenges}
%% ============================================================

\begin{frame}{Challenges Encountered}
  \begin{enumerate}
    \item \textbf{Low-data ML:}
          \begin{itemize}
            \item $<$150 labelled points $\Rightarrow$ high variance in CV metrics
            \item KNN outperforms deep methods -- typical in low-data settings
          \end{itemize}

    \vspace{4pt}
    \item \textbf{Pharmacophore–Scaffold Mismatch:}
          \begin{itemize}
            \item Phase hypothesis AHHR\_1 returns \textbf{zero hits} for all compounds
            \item Likely due to unusual 3-D geometry of the spiro-piperidine motif
            \item Phase screening with \texttt{-match 3} (loose) still gives no matches
          \end{itemize}

    \vspace{4pt}
    \item \textbf{Software Environment Inconsistencies:}
          \begin{itemize}
            \item Conda \texttt{qsar-env}: 210 RDKit features \; $\neq$ \;
                  artifact expects 217 $\Rightarrow$ always use \texttt{.venv}
            \item UMAP \texttt{check\_array(force\_all\_finite)} API mismatch
                  in conda env
          \end{itemize}

    \vspace{4pt}
    \item \textbf{Git Large-File Push Failure:}
          \begin{itemize}
            \item REINVENT4 \texttt{.prior} binaries ($\sim$2 GB) caused HTTP 500
            \item Resolved via \texttt{git reset --mixed} + \texttt{.gitignore} update
          \end{itemize}

    \vspace{4pt}
    \item \textbf{Non-Numeric IC\(_50\) Values:}
          \begin{itemize}
            \item Entries labelled \texttt{>100} or \texttt{<0.39}\;
                  handled as auxiliary set with standardised SMILES predictions
          \end{itemize}
  \end{enumerate}
